**Config-driven dashboard metrics (UX-06):**

Update Dashboard.jsx to display metrics derived from entity config, NOT hardcoded school-choice labels. The dashboard must work for any domain module.

1. Fetch the entity registry via `useQuery({ queryKey: ['entities'], queryFn: getEntities })` from the entities API (created in Plan 01)
2. For each entity with `auto_crud: true`, show a count metric card by fetching `getEntityList(entity.table)` to get the row count
3. For entities without `auto_crud` (like student), use existing domain-specific API calls (e.g., `getStudents()`) wrapped in useQuery
4. The metrics array is built dynamically from the entity registry response, not from a hardcoded array

```jsx
import { Card, CardHeader, CardTitle, CardContent } from '@/components/ui/card';
import { useQuery, useQueries } from '@tanstack/react-query';
import { getEntities, getEntityList } from '../../api/entities';
import { getStudents } from '../../api/students';

// Inside Dashboard:
const entitiesQuery = useQuery({ queryKey: ['entities'], queryFn: getEntities });

// Build metrics dynamically from entity config
const entityMetrics = (entitiesQuery.data ?? [])
  .filter(e => e.auto_crud)
  .map(entity => ({
    label: entity.name.charAt(0).toUpperCase() + entity.name.slice(1),
    queryKey: ['entities', entity.name, 'count'],
    queryFn: () => getEntityList(entity.table),
  }));

// Use useQueries for parallel entity count fetches
const entityCountQueries = useQueries({
  queries: entityMetrics.map(m => ({
    queryKey: m.queryKey,
    queryFn: m.queryFn,
    enabled: !!entitiesQuery.data,
  })),
});

// Domain-specific metrics (student is not auto_crud, so fetch separately)
const studentsQuery = useQuery({ queryKey: ['students'], queryFn: getStudents });

// Combine all metrics
const metrics = [
  { label: 'Total Students', value: studentsQuery.data?.length ?? '—' },
  { label: 'Active Students', value: studentsQuery.data?.filter(s => !s.graduated).length ?? '—' },
  ...entityMetrics.map((m, i) => ({
    label: m.label,
    value: entityCountQueries[i]?.data?.length ?? '—',
  })),
];

// Render metrics grid:
<div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
  {metrics.map((m) => (
    <Card key={m.label}>
      <CardHeader>
        <CardTitle style={{ fontSize: 'var(--font-size-sm)', fontWeight: 'var(--font-weight-medium)', color: 'var(--color-text-secondary)' }}>
          {m.label}
        </CardTitle>
      </CardHeader>
      <CardContent>
        <p style={{ fontSize: 'var(--font-size-2xl)', fontWeight: 'var(--font-weight-medium)', color: 'var(--color-text-primary)' }}>
          {m.value}
        </p>
      </CardContent>
    </Card>
  ))}
</div>
```

The key difference from a hardcoded approach: when a new entity is added via YAML, it automatically appears as a metric card on the dashboard without any frontend code changes.

**Create `frontend/src/components/TemplateSelector/TemplateSelector.jsx`:**

```jsx
import { Button } from '@/components/ui/button';

const DEFAULT_TEMPLATES = [
  { id: 'professional', label: 'Professional' },
  { id: 'modern', label: 'Modern' },
  { id: 'minimal', label: 'Minimal' },
];

export default function TemplateSelector({
  templateId,
  onTemplateChange,
  templates = DEFAULT_TEMPLATES,
  isPending = false,
}) {
  return (
    <div style={{ display: 'flex', gap: 'var(--space-2)', alignItems: 'center' }}>
      <span style={{
        fontSize: 'var(--font-size-sm)',
        fontWeight: 'var(--font-weight-medium)',
        color: 'var(--color-text-secondary)',
      }}>
        Template:
      </span>
      {templates.map((t) => (
        <Button
          key={t.id}
          variant={templateId === t.id ? 'default' : 'secondary'}
          size="sm"
          onClick={() => onTemplateChange(t.id)}
          disabled={isPending}
        >
          {t.label}
        </Button>
      ))}
    </div>
  );
}
```
