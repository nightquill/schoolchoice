**Pattern for each hook** (following useToast.js convention):

```javascript
// hooks/usePersonalTab.js
import { useState, useEffect, useCallback } from 'react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { updateStudent } from '../api/students';

export function usePersonalTab(student, studentId) {
  const queryClient = useQueryClient();
  const [form, setForm] = useState({ /* fields from student */ });
  const [errors, setErrors] = useState({});

  useEffect(() => {
    if (student) {
      setForm({ /* populate from student fields */ });
    }
  }, [student]);

  const mutation = useMutation({
    mutationFn: (payload) => updateStudent(studentId, payload),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['student', studentId] });
    },
    onError: () => { /* set error state */ },
  });

  const handleChange = useCallback((e) => {
    const { name, value, type, checked } = e.target;
    setForm((prev) => ({ ...prev, [name]: type === 'checkbox' ? checked : value }));
  }, []);

  const handleSave = useCallback(() => {
    mutation.mutate(form);
  }, [form, mutation]);

  return { form, saving: mutation.isPending, errors, handleChange, handleSave };
}
```

**Per-hook specifics extracted from StudentProfile.jsx:**

- **usePersonalTab**: Form state for student fields (full_name, chinese_name, class_name, etc.), save via updateStudent, graduation toggle via graduateStudent
- **useGradesTab**: Grade list management, add/delete grades via createGrade/deleteGrade, subject list fetch, transcript upload + polling. **CRITICAL per RESEARCH.md Pitfall 6:** polling interval must use `useRef` + `useEffect` cleanup: `return () => clearInterval(intervalRef.current)`
- **useLanguageTab**: Language proficiency form state, save via updateStudent
- **useEvaluationsTab**: Teacher evaluation entries management
- **useActivitiesTab**: Activities list management (add/remove/edit)
- **useNotesTab**: Notes CRUD operations
- **usePlansTab**: Plan generation trigger, plan list, plan status polling (same polling cleanup pattern as GradesTab)

Each hook:
1. Takes `(student, studentId)` as arguments (plus any other needed params like `showToast`)
2. Uses `useMutation` with `queryClient.invalidateQueries` on success (per D-10)
3. Returns an object with all state and handlers the tab component needs
4. Uses named export (not default): `export function useXxxTab(...)`

**CRITICAL v5 rule:** Do NOT use `onSuccess`/`onError` inside `useQuery`. These callbacks are only valid on `useMutation`.

**Rewrite StudentProfile.jsx parent** (per D-01):

The parent shrinks from ~1450 lines to ~60 lines:

```javascript
import { useParams, useSearchParams } from 'react-router-dom';
import { useQuery } from '@tanstack/react-query';
import { getStudent } from '../../api/students';
import NavBarV2 from '../../components/NavBarV2/NavBarV2';
import Tabs from '../../components/Tabs/Tabs';
import QueryBoundary from '../../components/QueryBoundary/QueryBoundary';
import PersonalTab from './PersonalTab';
import GradesTab from './GradesTab';
import LanguageTab from './LanguageTab';
import EvaluationsTab from './EvaluationsTab';
import ActivitiesTab from './ActivitiesTab';
import NotesTab from './NotesTab';
import PlansTab from './PlansTab';

const TABS = [
  { id: 'personal', label: 'Personal' },
  { id: 'grades', label: 'Grades' },
  { id: 'language', label: 'Language' },
  { id: 'evaluations', label: 'Teacher Evaluations' },
  { id: 'activities', label: 'Activities' },
  { id: 'notes', label: 'Notes' },
  { id: 'plans', label: 'Plans' },
];

export default function StudentProfile() {
  const { id } = useParams();
  const [searchParams, setSearchParams] = useSearchParams();
  const activeTab = searchParams.get('tab') || 'personal';

  const { data: student, isLoading, isError, error, refetch } = useQuery({
    queryKey: ['student', id],
    queryFn: () => getStudent(id),
  });

  const tabContent = {
    personal: <PersonalTab student={student} studentId={id} />,
    grades: <GradesTab student={student} studentId={id} />,
    language: <LanguageTab student={student} studentId={id} />,
    evaluations: <EvaluationsTab student={student} studentId={id} />,
    activities: <ActivitiesTab student={student} studentId={id} />,
    notes: <NotesTab student={student} studentId={id} />,
    plans: <PlansTab student={student} studentId={id} />,
  };

  return (
    <div style={{ background: 'var(--color-background)', minHeight: '100vh', fontFamily: 'var(--font-family-base)' }}>
      <NavBarV2 />
      <main style={{ padding: 'var(--space-6) var(--space-8)' }}>
        <QueryBoundary isLoading={isLoading} isError={isError} error={error} refetch={refetch} resourceName="student">
          <h1 style={{ fontSize: 'var(--font-size-2xl)', fontWeight: 'var(--font-weight-medium)', marginBottom: 'var(--space-4)' }}>
            {student?.full_name}
          </h1>
          <Tabs tabs={TABS} activeTab={activeTab} onChange={(tabId) => setSearchParams({ tab: tabId })}>
            {tabContent[activeTab]}
          </Tabs>
        </QueryBoundary>
      </main>
    </div>
  );
}
```
