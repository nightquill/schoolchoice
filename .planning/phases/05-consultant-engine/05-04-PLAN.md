import pytest
from unittest.mock import patch, AsyncMock, MagicMock

class TestConsultantStream:
    def test_stream_endpoint_returns_sse(self, client, auth_headers):
        """Verify the stream endpoint returns SSE content type"""
        async def mock_stream(*args, **kwargs):
            yield "data: hello\n\n"
            yield "data: world\n\n"

        with patch("app.api.v1.routes.consultant.call_ai_stream", side_effect=mock_stream):

                mock_task.max_tokens = 4096
                mock_task.temperature = 0.3
                MockEngine.return_value.load_task.return_value = mock_task
                MockEngine.return_value.build_messages.return_value = [{"role": "user", "content": "test"}]

                response = client.get(
                    "/api/v1/consultant/tasks/academic_plan/stream?entity_id=test-id",
                    headers=auth_headers,
                )
                assert response.status_code == 200
                assert "text/event-stream" in response.headers.get("content-type", "")

    def test_stream_requires_auth(self, client):
        response = client.get("/api/v1/consultant/tasks/academic_plan/stream?entity_id=test-id")

        """Verify stream endpoint accepts token as query parameter"""
        with patch("app.api.v1.routes.consultant.call_ai_stream") as mock_stream, \
             patch("app.api.v1.routes.consultant.TaskEngine") as MockEngine:

            mock_task = MagicMock()
            mock_task.max_tokens = 4096
            mock_task.temperature = 0.3
            MockEngine.return_value.load_task.return_value = mock_task
            MockEngine.return_value.build_messages.return_value = [{"role": "user", "content": "test"}]

            response = client.get(
                f"/api/v1/consultant/tasks/academic_plan/stream?entity_id=test-id&token={auth_token}",

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client -> Consultant API | Untrusted HTTP requests cross into authenticated API |
| AI output -> DB persist | AI-generated JSON validated before database write |
| EventSource -> Stream endpoint | Auth token passed as query param due to EventSource GET-only limitation |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-11 | Spoofing | consultant.py all endpoints | mitigate | Depends(get_current_user) on save/status endpoints; Depends(get_current_user_or_query_token) on stream endpoint; JWT verification in both paths |
| T-05-12 | Tampering | consultant.py save endpoint | mitigate | jsonschema + Pydantic double validation before DB write; confidence guardrail downgrades inflated tiers |
| T-05-13 | Denial of Service | consultant.py stream endpoint | mitigate | Rate limiting: 20 requests per entity per user per 24h; AI_TIMEOUT setting on LiteLLM call |
| T-05-14 | Tampering | consultant.py save endpoint XSS | mitigate | html.escape() filter applied in Jinja2 templates for all AI-generated text before HTML rendering |
| T-05-15 | Information Disclosure | query param token | accept | Token in URL is visible in browser history and server logs; acceptable for single-user deployment model per PROJECT.md; production deployments should use short-lived tokens |
</threat_model>
