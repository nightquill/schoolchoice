```javascript
import { useRef, useEffect } from 'react';

function SSEStreamDisplay({ tokens, isStreaming, error, onRetry }) {
  const containerRef = useRef(null);

  // Auto-scroll to bottom on new tokens
  useEffect(() => {
    if (containerRef.current && isStreaming) {
      containerRef.current.scrollTop = containerRef.current.scrollHeight;
    }
  }, [tokens, isStreaming]);
```

**Generate Plan handler:**
```javascript
const handleGenerate = useCallback(() => {
  setStreaming(true);
  setStreamTokens('');
  setStreamError(null);

  const token = localStorage.getItem('token');
  const url = `/api/v1/consultant/tasks/${taskId}/stream?entity_id=${id}&token=${token}`;
  const source = new EventSource(url);
  eventSourceRef.current = source;

  source.onmessage = (event) => {
    setStreamTokens((prev) => prev + event.data);
  };

  source.addEventListener('done', async () => {
    source.close();
    eventSourceRef.current = null;
    // Save the accumulated buffer
    try {
      const result = await saveConsultantTask(taskId, id, streamTokens);
      setPlan(result);
      showToast('Plan ready - view it here.', 'success');
    } catch (err) {
      setStreamError('Failed to save plan. Please try again.');
    }
    setStreaming(false);
  });

  source.onerror = () => {
    source.close();
    eventSourceRef.current = null;
    setStreamError('Generation was interrupted. Please try again.');
    setStreaming(false);
  };
}, [id, taskId, streamTokens, showToast]);
```

**Stop Generation handler:**

```javascript
const handleStopGeneration = useCallback(() => {

}, []);
```
