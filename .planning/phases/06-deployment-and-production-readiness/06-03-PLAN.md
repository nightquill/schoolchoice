**3. scripts/generate_secrets.sh** — Create new script (per D-07, DEP-06, RESEARCH Pattern 7):
```bash
#!/usr/bin/env bash
set -euo pipefail

OUT="${1:-.env}"

if ! command -v openssl &>/dev/null; then
  echo "Error: openssl not found. Install it or use: python3 -c \"import secrets; print(secrets.token_hex(32))\""
  exit 1
fi

SECRET_KEY=$(openssl rand -hex 32)

cat > "$OUT" <<EOF
# Generated by scripts/generate_secrets.sh — $(date -u +%Y-%m-%dT%H:%M:%SZ)
# Fill in CHANGE_ME values before starting the app.

# === REQUIRED (app will not start without these) ===
DATABASE_URL=CHANGE_ME
SECRET_KEY=${SECRET_KEY}

# === AUTH ===
ALGORITHM=HS256
ACCESS_TOKEN_EXPIRE_MINUTES=30

# === CORS ===
# Comma-separated list of allowed origins (your Vercel URL + any custom domain)
CORS_ORIGINS=CHANGE_ME

# === AI Provider (optional — AI features degrade gracefully without these) ===
AI_PROVIDER=gemini
AI_API_KEY=
# AI_MODEL=
# AI_BASE_URL=
# AI_TIMEOUT=30
EOF

echo "Generated: ${OUT}"
echo "  SECRET_KEY: set ($(echo -n "$SECRET_KEY" | wc -c | tr -d ' ') hex chars)"
echo ""
echo "  Next steps:"
echo "    1. Set DATABASE_URL to your Neon connection string (with ?sslmode=require)"
echo "    2. Set CORS_ORIGINS to your Vercel deployment URL"
echo "    3. Optionally set AI_API_KEY for AI features"
```

Make the script executable: `chmod +x scripts/generate_secrets.sh`

**1. .github/workflows/ci.yml** — Create CI/CD pipeline (per D-06, RESEARCH Pattern 6):

```yaml
name: CI/CD

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-python@v5
        with:
          python-version: "3.10"

      - name: Install backend dependencies
        run: pip install -r backend/requirements.txt

      - name: Run backend tests
        run: pytest backend/tests/ -x -q
        env:
          DATABASE_URL: "sqlite:///:memory:"
          SECRET_KEY: "test-secret-key-for-pytest-only-not-for-production"
          ALGORITHM: "HS256"
          ACCESS_TOKEN_EXPIRE_MINUTES: "30"
          CORS_ORIGINS: "http://localhost:5173"
          AI_PROVIDER: "gemini"
          AI_API_KEY: ""
          UPLOAD_DIR: "/tmp/test_uploads"
          PLAN_GENERATION_TIMEOUT_SECONDS: "30"

      - uses: actions/setup-node@v4
        with:
          node-version: "20"

      - name: Install frontend dependencies
        run: npm ci
        working-directory: frontend

      - name: Build frontend
        run: npm run build
        working-directory: frontend

      - name: Run frontend tests
        run: npx vitest run
        working-directory: frontend

  deploy-frontend:
    needs: test
    if: github.ref == 'refs/heads/main' && github.event_name == 'push'
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "20"

      - name: Install Vercel CLI
        run: npm install -g vercel

      - name: Deploy to Vercel
        run: vercel deploy --prod --token ${{ secrets.VERCEL_TOKEN }}
        working-directory: frontend
        env:
          VERCEL_ORG_ID: ${{ secrets.VERCEL_ORG_ID }}
          VERCEL_PROJECT_ID: ${{ secrets.VERCEL_PROJECT_ID }}

  deploy-backend:
    needs: test
    if: github.ref == 'refs/heads/main' && github.event_name == 'push'
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Install Railway CLI
        run: curl -fsSL https://railway.app/install.sh | sh

      - name: Deploy to Railway
        run: railway up --service backend
        working-directory: backend
        env:
          RAILWAY_TOKEN: ${{ secrets.RAILWAY_TOKEN }}
```

Create the `.github/workflows/` directory first: `mkdir -p .github/workflows`

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
